1. Take a number from user and check if its a even number or odd number.
Answer: 
#include<stdio.h>
int main(){
    int a;
    scanf("%d", &a);
    if(a%2 == 0){
        printf("Even number\n");
    }
    else{
        printf("Odd number\n");
    }
    return 0;
}

2. Take a number from user and check if its a positive or negative number.
Answer: 
#include<stdio.h>
int main(){
    int a;
    scanf("%d", &a);
    if(a> 0){
        printf("Positive number\n");
    }
    else if(a< 0){
        printf("Negative number\n");
    }
    else{
        printf("Zero\n");
    }
    return 0;
}

First, the if condition is checked.

If it is false, then the program goes to the else if block.
